Create a new file: `src/components/SmoothScroll.jsx`

```jsx
import { useEffect, useRef } from 'react'
import Lenis from 'lenis'
import 'lenis/dist/lenis.css' // Import the default styles

export default function SmoothScroll({ children }) {
  useEffect(() => {
    // 1. Initialize Lenis
    const lenis = new Lenis({
      duration: 1.2,      // How long the scroll animation lasts
      easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)), // Smooth "out" easing
      smoothWheel: true,  // Enable for mouse wheel
      wheelMultiplier: 1, // Sensitivity
    })

    // 2. The Request Animation Frame (RAF) loop
    // This makes the scroll frame-rate independent and buttery smooth
    function raf(time) {
      lenis.raf(time)
      requestAnimationFrame(raf)
    }

    requestAnimationFrame(raf)

    // 3. Cleanup on unmount
    return () => {
      lenis.destroy()
    }
  }, [])

  return <>{children}</>
}
```

# index css
```css
/* Ensure the scroll feels natural and handles content correctly */
html.lenis, html.lenis body {
  height: auto;
}

.lenis.lenis-smooth {
  scroll-behavior: auto !important;
}

.lenis.lenis-smooth [data-lenis-prevent] {
  overscroll-behavior: contain;
}

.lenis.lenis-stopped {
  overflow: hidden;
}
```
