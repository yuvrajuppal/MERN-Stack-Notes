```jsx
import { useState } from "react";

function ColorPicker() {
  const [color, setColor] = useState("#ff0000");

  return (
    <div>
      <input
        type="color"
        value={color}
        onChange={(e) => setColor(e.target.value)}
      />
      <p>Selected: {color}</p>

      <div
        style={{
          width: 60,
          height: 30,
          backgroundColor: color,
          border: "1px solid #ccc",
          marginTop: 8,
        }}
      />
    </div>
  );
}
```
