import React from 'react'
import style from './card.module.css'

	let dogimage = `https://plus.unsplash.com/premium_photo-1694819488591-a43907d1c5cc?fm=jpg&q=60&w=3000&ixlib=rb-4.1.0&ixid=M3wxMjA3fDB8MHxzZWFyY2h8MXx8Y3V0ZSUyMGRvZ3N8ZW58MHx8MHx8fDA%3D`

	let dogage = 6
	return (
	<div className={style.card}>
		<img className={style.cardimage} src={dogimage} />
		<div>dog name - {dogname}</div>
		<div>age age - {dogage}</div>
	</div>
	)
}
export default Card

import React, { useState } from 'react'
function App() {

	return (
		<>

				height: '100px',
				width: '100px',
				backgroundColor: color
			}}>
			</div>
			
			<button onClick={() => {
				set_color("red")
			}}>click me</button>
			
		</>
	)

}
