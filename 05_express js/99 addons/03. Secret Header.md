```js
const API_SECRET_KEY = "my_super_secret_mobile_key_123";

app.use((req, res, next) => {
  const clientKey = req.header('X-App-Service-Key');
  if (clientKey === API_SECRET_KEY) {
    next();
  } else {
    res.status(403).json({ error: "Access Denied: Unrecognized Client" });
  }
});
```

# axios
```js
import axios from 'axios';

// Replace with your computer's IP if testing locally!
const API_BASE_URL = 'http://192.168.1.XX:5000'; 

const apiClient = axios.create({
  baseURL: API_BASE_URL,
  headers: {
    'Content-Type': 'application/json',
    'X-App-Service-Key': 'my_super_secret_mobile_key_123' // The "Lock"
  },
});

export default apiClient;
```

```js
import apiClient from './api';

const fetchData = async () => {
  try {
    const response = await apiClient.get('/api/data');
    console.log(response.data);
  } catch (error) {
    console.error("Access denied or server down:", error);
  }
};
```
