```js


import mysql from 'mysql2/promise'


const mysqlinfo = {
    host: 'localhost',
    user: 'root',
    password: '',
    database: 'wallpaperapp',
}
const checkdbconnect = async() =>{
    try{
        const connection = await mysql.createConnection(mysqlinfo);
        console.log("Mysql Db Connected");
         connection.end();  
    }catch(err){
        console.log(err);
    }
}
checkdbconnect()
```
