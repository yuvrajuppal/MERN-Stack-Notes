const mysqlinfo = {
    host: 'localhost',
    user: 'root',
    password: '',
    database: 'wallpaperapp',
}

const pool = mysql.createPool(mysqlinfo)

app.post('/add_Wallpaper', async (req, res) => {
    try {
        const { id, wallaper_name, description, imageLink } = req.body;
        if (!id || !wallaper_name || !description || !imageLink) {
            res.status(201).json({
                success: false,
                message: 'enter all values'
            })
        }
        const qry = `INSERT INTO wallpaper_info (id,wallaper_name,description,imageLink) VALUES (?,?,?,?)`
        const result = await pool.execute(qry, [id, wallaper_name, description, imageLink]);
        res.status(200).json({
            success: true,
            result
        })
    } catch (err) { 
       console.log(err);
        res.status(400).json({
            success: false,
            message: err.message
        })
    }
})
