import { v4 as uuidv4 } from 'uuid';
import { jtwsecurekey, pool, productmode } from '../configs/dbconfig.js';
import jwt from 'jsonwebtoken';

export const signupuser = async (req, res) => {
try {
  
const uuid = uuidv4();
const { email, password } = req.body;
const sql = `insert into userloginingo (id,email,password)
values(?,?,?)`
const result = await pool.execute(sql, [uuid, email, password])
const userinfo = {
uuid: uuid,
email: email
}
const token = jwt.sign(userinfo, jtwsecurekey)  
res.cookie('user_token', token, {
httpOnly: true,
secure: productmode,
sameSite: productmode ? 'None' : 'Lax',
maxAge: 1000 * 60 * 60 * 24 * 7
})
return res.status(200).json({
success: true,
result: result,
token
})
} catch (err) {
return res.status(400).json({ error: err.message })
}
}

  
  
export const userlogin = async (req, res) => {
try {
	const { email, password } = req.body;
	const sql = `select * from userloginingo where email = ? and password =?`
	const [result] = await pool.execute(sql, [email, password])
	if (result.length > 0) {
	const userdata = result[0]
	const userinfo = {
	uuid: userdata.id,
	email: userdata.email
	}
	const token = jwt.sign(userinfo, jtwsecurekey)
	res.cookie('user_token', token, {
	httpOnly: true,
	secure: productmode,
	sameSite: productmode ? 'None' : 'Lax',
	maxAge: 1000 * 60 * 60 * 24 * 7
	})
	return res.status(200).json({
	success: true,
	result: result
	})
	}
	return res.status(401).json({
	success: false,
	message: "Wrong email or password"
	})
} catch (err) {
res.status(400).json({ error: err.message })
}
}


export const checkuserislogin = async (req, res) => {
try {
const { user_token } = req.cookies
if(!user_token){
return res.status(401).json({
success:false,
message:'not token '
})
}
const decodedata = jwt.verify(user_token,jtwsecurekey);
const {uuid,email} = decodedata;
const sql = `select * from userloginingo where id = ?`
const [result] = await pool.execute(sql,[uuid]);
if(result.length>0){
return res.status(200).json({
success:true,
message:'valid user',

email:email
})
}
return res.status(401).json({
