allow all origin in app.js
```js
import cookieParser from 'cookie-parser'

app.use(cookieParser());

app.use(
	cors(
		{
		origin: (origin, callback) => {
		if (!origin) {
		return callback(null, true)
		}
		return callback(null, origin)
		},
		credentials: true,
		}
	)
)
```
