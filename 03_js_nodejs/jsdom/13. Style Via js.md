
## way 1
```html
<p id="para">Hello, this is a paragraph</p>
<button id="changeBtn">Change Style</button>
```

```js
const para = document.getElementById('para');
const btn = document.getElementById('changeBtn');

btn.addEventListener('click', () => {
  para.style.color = 'blue';             // text color
  para.style.fontSize = '20px';          // font size
  para.style.backgroundColor = '#f0f0f0'; // background color
  para.style.padding = '10px';           // padding
  para.style.border = '2px solid red';   // border
});
```

## way 2

```js
para.style.cssText = `
  color: blue;
  font-size: 20px;
  background-color: #f0f0f0;
  padding: 10px;
  border: 2px solid red;
`;
```

## way 3
```js
para.style.setProperty('color', 'blue');
para.style.setProperty('font-size', '20px');
para.style.setProperty('background-color', '#f0f0f0');

// With !important
para.style.setProperty('display', 'block', 'important');

```
