```js
const text = "Hello World";

// First 5 characters
console.log(text.slice(0, 5)); // "Hello" [web:2]

// From index 6 to end
console.log(text.slice(6)); // "World" [web:2]

// Last 5 characters
console.log(text.slice(-5)); // "World" [web:2]

// Extract using indexOf for dynamic slicing
const email = "user@example.com";
const username = email.slice(0, email.indexOf("@")); // "user" [web:9]

```
