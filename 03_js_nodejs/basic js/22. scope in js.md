```js
// var leaks outside block
if (true) {
    var x = 10;
}
console.log(x); // 10 ✅

// let stays inside block
if (true) {
    let y = 20;
}
console.log(y); // ReferenceError ❌ 

```

```js
console.log(a); // undefined ✅
var a = 5;

console.log(b); // ReferenceError ❌ (Temporal Dead Zone)
let b = 10; 
```
