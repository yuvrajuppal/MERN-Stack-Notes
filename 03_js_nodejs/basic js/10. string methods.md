```js
let mystring = "hello world banana"

// the all are ussed to target the string charactors
console.log(mystring[0]);
console.log(mystring.at(0));
console.log(mystring.charAt(0));

// this will show the charactor code of the string
console.log(mystring.codePointAt(0));

let mystring2 = "hello world"

// this will show the concatination of the string
console.log(mystring2.concat(" ", mystring));

// this will tell the end of the string is true or false
console.log(mystring2.endsWith("world"));

// this will tell the start of the string is true or false
console.log(mystring2.startsWith("hello"));

// this will tell the search string is true or false
console.log(mystring2.includes("hello"));

// this will search from start and tell the index of the string
console.log(mystring.indexOf("na"));

// this will search from end and tell the index of the string
console.log(mystring.lastIndexOf("na"));
  
// this will compare the string and tell the index of the string
console.log(mystring.localeCompare("na"));

let myname = "goosecode"

// this will remove the extra space from the string
console.log(myname.trim());
console.log(myname.trimStart());
console.log(myname.trimEnd());

// this will convert the string to uppercase
console.log(myname.toUpperCase());
  
// this will convert the string to lowercase
console.log(myname.toLowerCase());


// this will repeat the string
console.log(myname.repeat(2));

// this will replace the string
let number = "13-12323-12312312-1232"
console.log(number.replaceAll("-", ""));

// this will replace the first match
console.log('--------------------');
console.log(number.replace("-", ""));

// this will add the string at the start as padding
console.log(number.padStart(50, "0"));
  
// this will add the string at the end as padding
console.log(number.padEnd(50, "0"));
```
