`async` and `await` are JavaScript keywords that let you write asynchronous code (like API calls, file reads, timers, etc.) in a simple, clean, synchronous style, without deep `.then()` chains or callback hell


```js
async function getUser() {
  try {
    // Wait for the response
    const response = await fetch('https://jsonplaceholder.typicode.com/users/1');
    // Wait for JSON parsing
    const userData = await response.json();    
    console.log("User data:", userData);
    return userData;
  } catch (error) {
    console.error("Error fetching user:", error);
  }
}
// Call the async function
getUser();
```

Here’s a simple example of using `fetch` to get JSON data from an API:

```js
async function fetchUserData() {
  try {
    // Step 1: Send the request
    const response = await fetch("https://jsonplaceholder.typicode.com/users/1");
    // Step 2: Check if the response is OK (status 200–299)
    if (!response.ok) {
      console.log("crashed")
    }
    // Step 3: Parse the response as JSON
    const userData = await response.json();
    console.log("User data:", userData);
    return userData;
  } catch (error) {
    console.error("Something went wrong:", error.message);
  }
}
// Call the function
fetchUserData();
```

## Get and Post API
```js
// GET request: fetch a post
async function getPost() {
  try {
    const response = await fetch("https://jsonplaceholder.typicode.com/posts/1");
    if (!response.ok) {
      console.log(`Failed to fetch post:`)
    }
    const post = await response.json();
    console.log("Post:", post);
    return post;
  } catch (error) {
    console.error("Error in getPost:", error.message);
  }
}

// POST request: create a new post
const createPost = async () => {
  try {
    const response = await fetch("https://jsonplaceholder.typicode.com/posts", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        title: "My new post",
        body: "This is the body of my post",
        userId: 1,
      }),
    });
    if (!response.ok) {
      throw new Error(`Failed to create post: 
      ${response.status}`);
    }

    const newPost = await response.json();
    console.log("Created post:", newPost);
    return newPost;
  } catch (error) {
    console.error("Error in createPost:", error.message);
  }
}

// Usage
getPost();
createPost();

```
